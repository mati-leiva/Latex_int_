        &= 2\int_{0}^{\infty} \frac{x}{e^x - e^{-x}}dx\cdot \frac{e^{-x}}{e^{-x}}\\

        &= -2\int_{0}^{1} \frac{\ln(u)}{1 - u^2}du = -2\sum_{n=1}^{\infty}\int_{0}^{1}u^{2n}\ln(u)du\\
    \end{align*}
